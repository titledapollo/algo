int part(int a[],int lb,int ub)
{
int p,i,j,t1,t2;
i=lb-1;
//j=lb;
p=a[ub];
for(j=lb;j<ub;j++)
{
if(a[j]<=p)
{
i=i+1;
t1=a[j];
a[i]=a[j];
a[j]=t1;
}
}
t2=a[i+1];
a[i+1]=a[ub];
a[ub]=t2;
return i+1;
}
void qs(int a[],int lb,int ub)
{
int pvt;
if(lb<ub){
pvt=part(a,lb,ub);
qs(a,lb,(pvt-1));
qs(a,(pvt+1),ub);
}
}
